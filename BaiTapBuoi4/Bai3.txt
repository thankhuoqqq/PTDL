#Bài 3: Nhập các dãy số

#1) Gồm các số tự nhiên từ 1 đến 100
(1:100)

#2) Gồm các số tự nhiên từ 100 đến 1
(100:1)

#3) Gồm các số thập phân từ 10 đến 15, hai số liên tiếp hơn kém nhau 0.5
seq(10, 15, 0.5)

#4) Gồm các số thập phân từ 16 đến 10, dãy có 11 số
seq(length = 11, from = 16, to = 10)

#5) Gồm các số tự nhiên từ 2 đến 5, mỗi số tự nhiên xuất hiện liên tiếp 4 lần
rep(c(2:5), 4)
